# 最小生成树
节点数为n,边数为m的图。
## Prime算法
   `prime` 算法的算法思想与`dijkstra`相似
- 朴素版Prime
	算法复杂度 O(n^2)
	算法模板

	```cpp
	class Prime {
	public :
	    static int getMST(int n, std::vector<std::vector<int>> &a) {
	        std::vector<int> dis(n + 1, INF);
	        std::vector<bool> f(n + 1, false);
	        dis[1] = 0;
	        int res = 0;
	        for (int i = 0; i < n; i++) {
	            int pos = -1;
	            for (int j = 1; j <= n; j++)
	                if (!f[j] && (pos == -1 || dis[pos] > dis[j]))
	                    pos = j;
	            f[pos] = true;
	            res += dis[pos];
	            if (i && dis[pos] == INF)
	                return INF + INF;
	            for (int j = 1; j <= n; j++)
	                dis[j] = std::min(dis[j], a[pos][j]);
	        }
	        return res;
	    }
	};
	```

## Kurskal算法
- Kruskal算法
	算法复杂度O(mlogm)
	模板

```cpp
class Kruskal {
public :
    struct E {
        int u, v, w;

        bool friend operator<(const E &a, const E &b) {
            return a.w < b.w;
        }
    };

    int n{}, m{};
    std::vector<E> a;
    std::vector<int> f;

    void init() {
        std::cin >> n >> m;
        a.assign(m, {0, 0, 0});
        for (int i = 0; i < m; i++) {
            int u, v, w;
            std::cin >> u >> v >> w;
            a[i] = {u, v, w};
        }
    }

    int find(int x) {
        return x == f[x] ? x : f[x] = find(f[x]);
    }

    int GetMST() {
        int res = 0, cnt = 0;
        f.assign(n + 1, 0ll);
        std::iota(f.begin(), f.end(), 0ll);
        std::sort(a.begin(), a.end());
        for (int i = 0; i < m; i++) {
            auto [u, v, w] = a[i];
            int x = find(u), y = find(v);
            if (x != y) {
                cnt++, res += w;
                f[y] = x;
            }
        }
        if (cnt != n - 1)
            return INF + INF;
        return res;
    }
};
```

# 二分图

二分图:节点由两个集合组成，且两个集合内部没有边的图
二分图性质:当且仅当图中不含奇数环
## 染色法
- 染色法(判断是否为二分图)
	算法复杂度(n + m)
	 模板

```cpp
class drawBinaryGraph {
public:
    int n{}, m{};
    std::vector<std::vector<int>> a;
    std::vector<int> color;

    void init() {
        std::cin >> n >> m;
        a.assign(n + 1, std::vector<int>());
        color.assign(n + 1, 0);
        for (int i = 0; i < m; i++) {
            int u, v;
            std::cin >> u >> v;
            a[u].push_back(v);
            a[v].push_back(u);
        }
    }

    bool dfs(int u, int c) {
        color[u] = c;
        for (int i = 0; i < a[u].size(); i++) {
            int t = a[u][i];
            if (!color[t]) {
                if (!dfs(t, 3 - c))
                    return false;
            } else if (color[t] == c)
                return false;
        }
        return true;
    }

    bool isBinaryGraph() {
        bool t = true;
        for (int i = 1; i <= n; i++)
            if (!color[i])
                if (!dfs(i, 1)) {
                    t = false;
                    break;
                }
        return t;
    }
};
```

## 匈牙利算法
- 匈牙利算法
	算法复杂度最坏为O(mn), 实际运行时间小于O(mn)
	 模板

```cpp
	 class Hungary {
	public:
	int n1, n2, m;
	std::vector<std::vector<int>> l;
	std::vector<int> r;
	std::vector<bool> f;

	void init() {
		std::cin >> n1 >> n2 >> m;
		l.assign(n1 + 1, std::vector<int>());
		r.assign(n2 + 1, 0ll);
		f.assign(n2 + 1, false);
		for(int i = 0; i < m; i++) {
			int u, v;
			std::cin >> u >> v;
			l[u].push_back(v);
		}
	}

	bool find(int x) {
		for(int i = 0; i < l[x].size(); i++) {
			int j = l[x][i];
			if(!f[j]) {
				f[j] = true;
				if(r[j] == 0 || find(r[j])) {
					r[j] = x;
					return true;
				}
			}
		}
		return false;
	}

	int maxMatch() {
		int ans = 0;
		for(int i = 1; i <= n1; i++) {
			std::fill(f.begin(), f.end(), false);
			if(find(i))
				ans++;
		}
		return ans;
	}
};
```

# LCA(最近公共祖先)
## Tarjan算法
### 算法逻辑
算法前置学习：==并查集==，==邻接表==， ==树==

### 模板
[视频讲解](https://www.bilibili.com/video/BV1A94y12737/?spm_id_from=333.337.search-card.all.click&vd_source=8f15424b5c0558da6e1955b00bc12219)

```cpp
#include<bits/stdc++.h>  
  
#define int long long  
#define endl '\n'  
#define pii std::pair<int ,int>  
#define fix(x) std::fixed << std::setprecision(x)  
const int N = 1e7 + 50;  
  
int n, m, s;//n个点，m次询问，根节点s
std::vector<std::vector<pii > > query;//维护查询  
std::vector<std::vector<int> > e; //存图
std::vector<int> fa, vis, lca; //第i个节点的父节点fa[i],访问状态vis[i]
					//第i次询问的结果lca[i]
  
int find(int x) {  
    return fa[x] == x ? x : fa[x] = find(fa[x]);  
}  
  
void Tarjan(int u) {  
    vis[u] = 1;  
    for (auto v: e[u]) {  
        if (!vis[v]) {  
            Tarjan(v);  
            fa[v] = u;  
        }  
    }  
  
    for (auto [v, i]: query[u]) {  
        if (vis[v])  
            lca[i] = find(v);  
    }  
}  
  
void solve() {  
    std::cin >> n >> m >> s;  
    e = std::vector<std::vector<int> >(n + 1);  
    query = std::vector<std::vector<pii > >(n + 1);  
    fa = std::vector<int>(n + 1);  
    vis = std::vector<int>(n + 1, 0);  
    lca = std::vector<int>(m + 1, -1);  
    std::iota(fa.begin(), fa.end(), 0ll);  
    for (int i = 1, u, v; i <= n - 1; i++) {  
        std::cin >> u >> v;  
        e[u].push_back(v);  
        e[v].push_back(u);  
    }  
  
    for (int i = 1, u, v; i <= m; i++) {  
        std::cin >> u >> v;  
        query[u].push_back({v, i});  
        query[v].push_back({u, i});  
    }  
    Tarjan(s);  
  
    for (int i = 1; i <= m; i++)  
        std::cout << lca[i] << endl;  
}  
  
signed main() {  
    std::ios::sync_with_stdio(false);  
    std::cin.tie(nullptr), std::cout.tie(nullptr);  
    int TT = 1;  
//    std::cin >> TT;  
    while (TT--) solve();  
    return 0;  
}
```
